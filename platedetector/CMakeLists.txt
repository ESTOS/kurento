include_directories(
  ${GSTREAMER_INCLUDE_DIRS}
  ${GSTREAMER_VIDEO_INCLUDE_DIRS}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${OPENCV_INCLUDE_DIRS}
  ${TESSERACT_INCLUDE_DIRS}
)

  platedetector.c
  kmsplatedetector.c kmsplatedetector.h

  ${GSTREAMER_VIDEO_LIBRARIES}
  ${OPENCV_LIBRARIES}
  ${TESSERACT_LIBRARIES}
)
